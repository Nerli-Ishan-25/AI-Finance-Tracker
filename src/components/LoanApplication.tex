import React, { useState } from 'react';
import { ApplicationForm } from './loan/ApplicationForm';
import { RiskAssessment } from './loan/RiskAssessment';
import { ExplainabilityPanel } from './loan/ExplainabilityPanel';

export function LoanApplication() {
  const [applicationData, setApplicationData] = useState<any>(null);
  const [riskScore, setRiskScore] = useState<number | null>(null);
  const [isAnalyzing, setIsAnalyzing] = useState(false);

  const handleSubmit = async (data: any) => {
    setIsAnalyzing(true);
    setApplicationData(data);
    
    // Simulate ML model prediction
    setTimeout(() => {
      const mockRiskScore = Math.random() * 0.4 + 0.1; // Random score between 0.1 and 0.5
      setRiskScore(mockRiskScore);
      setIsAnalyzing(false);
    }, 2000);
  };

  const resetAssessment = () => {
    setApplicationData(null);
    setRiskScore(null);
    setIsAnalyzing(false);
  };

  return (
    <div className="space-y-6">
      <div className="flex items-center justify-between">
        <h1 className="text-3xl font-bold text-gray-900">Loan Risk Assessment</h1>
        {applicationData && (
          <button
            onClick={resetAssessment}
            className="px-4 py-2 text-sm font-medium text-gray-600 hover:text-gray-900"
          >
            New Assessment
          </button>
        )}
      </div>

      {!applicationData ? (
        <ApplicationForm onSubmit={handleSubmit} />
      ) : (
        <div className="grid grid-cols-1 lg:grid-cols-2 gap-6">
          <div className="space-y-6">
            <RiskAssessment 
              applicationData={applicationData}
              riskScore={riskScore}
              isAnalyzing={isAnalyzing}
            />
          </div>
          
          <div>
            <ExplainabilityPanel 
              applicationData={applicationData}
              riskScore={riskScore}
              isAnalyzing={isAnalyzing}
            />
          </div>
        </div>
      )}
    </div>
  );
}
