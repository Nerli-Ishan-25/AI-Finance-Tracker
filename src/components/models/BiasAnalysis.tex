import React from 'react';
import { AlertTriangle, CheckCircle, Users } from 'lucide-react';

export function BiasAnalysis() {
  const biasMetrics = [
    {
      category: 'Age Groups',
      groups: ['18-25', '26-35', '36-50', '51-65', '65+'],
      approvalRates: [72.3, 78.9, 82.1, 79.4, 76.8],
      status: 'compliant',
      maxDifference: 9.8,
    },
    {
      category: 'Gender',
      groups: ['Male', 'Female'],
      approvalRates: [78.2, 79.1],
      status: 'compliant',
      maxDifference: 0.9,
    },
    {
      category: 'Geographic Regions',
      groups: ['Northeast', 'Southeast', 'Midwest', 'West', 'Southwest'],
      approvalRates: [81.2, 73.4, 79.6, 82.8, 75.9],
      status: 'warning',
      maxDifference: 9.4,
    },
    {
      category: 'Income Brackets',
      groups: ['<$50k', '$50k-$100k', '$100k-$150k', '>$150k'],
      approvalRates: [65.2, 79.8, 88.7, 92.1],
      status: 'compliant',
      maxDifference: 26.9,
    },
  ];

  const getStatusIcon = (status: string) => {
    switch (status) {
      case 'compliant':
        return <CheckCircle className="h-5 w-5 text-green-500" />;
      case 'warning':
        return <AlertTriangle className="h-5 w-5 text-yellow-500" />;
      default:
        return <AlertTriangle className="h-5 w-5 text-red-500" />;
    }
  };

  const getStatusColor = (status: string) => {
    switch (status) {
      case 'compliant':
        return 'text-green-700 bg-green-100';
      case 'warning':
        return 'text-yellow-700 bg-yellow-100';
      default:
        return 'text-red-700 bg-red-100';
    }
  };

  return (
    <div className="space-y-6">
      <div className="bg-white rounded-lg shadow-sm border">
        <div className="p-6">
          <div className="flex items-center space-x-2 mb-6">
            <Users className="h-5 w-5 text-blue-600" />
            <h2 className="text-xl font-semibold text-gray-900">Bias Analysis Report</h2>
          </div>

          <div className="mb-6 p-4 bg-blue-50 rounded-lg">
            <h3 className="font-medium text-blue-900 mb-2">Fair Lending Compliance</h3>
            <p className="text-sm text-blue-800">
              This analysis tests for disparate impact across protected classes as required by the 
              Equal Credit Opportunity Act (ECOA) and Fair Housing Act (FHA). A difference in approval 
              rates exceeding 20% may indicate potential bias requiring investigation.
            </p>
          </div>

          <div className="space-y-6">
            {biasMetrics.map((metric, index) => (
              <div key={index} className="border rounded-lg p-4">
                <div className="flex items-center justify-between mb-4">
                  <div className="flex items-center space-x-3">
                    {getStatusIcon(metric.status)}
                    <h4 className="font-medium text-gray-900">{metric.category}</h4>
                  </div>
                  <div className="flex items-center space-x-3">
                    <span className="text-sm text-gray-500">
                      Max Difference: {metric.maxDifference.toFixed(1)}%
                    </span>
                    <span className={`px-3 py-1 text-sm font-medium rounded-full ${getStatusColor(metric.status)}`}>
                      {metric.status === 'compliant' ? 'Compliant' : 'Review Required'}
                    </span>
                  </div>
                </div>

                <div className="grid grid-cols-2 md:grid-cols-3 lg:grid-cols-5 gap-4">
                  {metric.groups.map((group, groupIndex) => (
                    <div key={groupIndex} className="text-center p-3 bg-gray-50 rounded-lg">
                      <div className="text-sm font-medium text-gray-900">{group}</div>
                      <div className="text-lg font-bold text-gray-900 mt-1">
                        {metric.approvalRates[groupIndex]}%
                      </div>
                      <div className="w-full bg-gray-200 rounded-full h-2 mt-2">
                        <div
                          className="bg-blue-500 h-2 rounded-full"
                          style={{ width: `${metric.approvalRates[groupIndex]}%` }}
                        ></div>
                      </div>
                    </div>
                  ))}
                </div>

                {metric.status === 'warning' && (
                  <div className="mt-4 p-3 bg-yellow-50 border border-yellow-200 rounded-lg">
                    <p className="text-sm text-yellow-800">
                      <strong>Action Required:</strong> Geographic approval rate differences detected. 
                      Review regional economic factors and ensure consistent underwriting standards.
                    </p>
                  </div>
                )}
              </div>
            ))}
          </div>
        </div>
      </div>

      <div className="bg-white rounded-lg shadow-sm border">
        <div className="p-6">
          <h3 className="text-lg font-semibold text-gray-900 mb-4">Bias Mitigation Strategies</h3>
          
          <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
            <div className="p-4 border rounded-lg">
              <h4 className="font-medium text-gray-900 mb-2">Pre-processing</h4>
              <ul className="text-sm text-gray-600 space-y-1">
                <li>• Remove protected attributes from training data</li>
                <li>• Use synthetic data generation for underrepresented groups</li>
                <li>• Apply fairness-aware feature selection</li>
              </ul>
            </div>
            
            <div className="p-4 border rounded-lg">
              <h4 className="font-medium text-gray-900 mb-2">In-processing</h4>
              <ul className="text-sm text-gray-600 space-y-1">
                <li>• Implement fairness constraints during training</li>
                <li>• Use adversarial debiasing techniques</li>
                <li>• Apply demographic parity regularization</li>
              </ul>
            </div>
            
            <div className="p-4 border rounded-lg">
              <h4 className="font-medium text-gray-900 mb-2">Post-processing</h4>
              <ul className="text-sm text-gray-600 space-y-1">
                <li>• Adjust decision thresholds by group</li>
                <li>• Apply calibration across protected classes</li>
                <li>• Use equalized odds correction</li>
              </ul>
            </div>
            
            <div className="p-4 border rounded-lg">
              <h4 className="font-medium text-gray-900 mb-2">Monitoring</h4>
              <ul className="text-sm text-gray-600 space-y-1">
                <li>• Continuous bias monitoring in production</li>
                <li>• Regular model audits and retraining</li>
                <li>• Automated alerting for bias threshold breaches</li>
              </ul>
            </div>
          </div>
        </div>
      </div>
    </div>
  );
}
