import React from 'react';
import { BarChart3, FileText, Brain, Shield, Building2 } from 'lucide-react';

interface NavigationProps {
  activeTab: 'dashboard' | 'application' | 'models' | 'compliance';
  setActiveTab: (tab: 'dashboard' | 'application' | 'models' | 'compliance') => void;
}

export function Navigation({ activeTab, setActiveTab }: NavigationProps) {
  const tabs = [
    { id: 'dashboard', label: 'Risk Dashboard', icon: BarChart3 },
    { id: 'application', label: 'Loan Assessment', icon: FileText },
    { id: 'models', label: 'Model Analysis', icon: Brain },
    { id: 'compliance', label: 'Compliance Monitor', icon: Shield },
  ] as const;

  return (
    <nav className="bg-white shadow-sm border-b">
      <div className="container mx-auto px-4">
        <div className="flex items-center justify-between h-16">
          <div className="flex items-center space-x-2">
            <Building2 className="h-8 w-8 text-blue-600" />
            <span className="text-xl font-bold text-gray-900">FinRisk AI</span>
          </div>
          
          <div className="flex space-x-1">
            {tabs.map((tab) => {
              const Icon = tab.icon;
              const isActive = activeTab === tab.id;
              
              return (
                <button
                  key={tab.id}
                  onClick={() => setActiveTab(tab.id)}
                  className={`flex items-center space-x-2 px-4 py-2 rounded-lg text-sm font-medium transition-colors ${
                    isActive
                      ? 'bg-blue-100 text-blue-700'
                      : 'text-gray-600 hover:text-gray-900 hover:bg-gray-100'
                  }`}
                >
                  <Icon className="h-4 w-4" />
                  <span>{tab.label}</span>
                </button>
              );
            })}
          </div>
        </div>
      </div>
    </nav>
  );
}
