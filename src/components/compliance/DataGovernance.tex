import React from 'react';
import { Database, Shield, Eye, AlertTriangle } from 'lucide-react';

export function DataGovernance() {
  const dataMetrics = [
    {
      category: 'Data Quality',
      score: 96.2,
      status: 'excellent',
      details: 'Missing values < 2%, outliers identified and handled',
    },
    {
      category: 'Data Security',
      score: 98.7,
      status: 'excellent',
      details: 'Encryption at rest and in transit, access controls active',
    },
    {
      category: 'Data Privacy',
      score: 94.3,
      status: 'good',
      details: 'PII anonymization implemented, consent tracking active',
    },
    {
      category: 'Data Lineage',
      score: 91.8,
      status: 'good',
      details: 'Complete data flow documentation, source tracking enabled',
    },
  ];

  const getStatusIcon = (status: string) => {
    switch (status) {
      case 'excellent':
        return <Shield className="h-5 w-5 text-green-500" />;
      case 'good':
        return <Eye className="h-5 w-5 text-blue-500" />;
      case 'warning':
        return <AlertTriangle className="h-5 w-5 text-yellow-500" />;
      default:
        return <AlertTriangle className="h-5 w-5 text-red-500" />;
    }
  };

  const getStatusColor = (status: string) => {
    switch (status) {
      case 'excellent':
        return 'text-green-700 bg-green-100';
      case 'good':
        return 'text-blue-700 bg-blue-100';
      case 'warning':
        return 'text-yellow-700 bg-yellow-100';
      default:
        return 'text-red-700 bg-red-100';
    }
  };

  return (
    <div className="bg-white rounded-lg shadow-sm border">
      <div className="p-6">
        <div className="flex items-center space-x-2 mb-6">
          <Database className="h-5 w-5 text-blue-600" />
          <h2 className="text-xl font-semibold text-gray-900">Data Governance</h2>
        </div>

        <div className="space-y-4 mb-6">
          {dataMetrics.map((metric, index) => (
            <div key={index} className="flex items-center justify-between p-4 border rounded-lg">
              <div className="flex items-center space-x-3">
                {getStatusIcon(metric.status)}
                <div>
                  <h4 className="font-medium text-gray-900">{metric.category}</h4>
                  <p className="text-sm text-gray-600">{metric.details}</p>
                </div>
              </div>
              
              <div className="text-right">
                <div className="text-xl font-bold text-gray-900">{metric.score}%</div>
                <span className={`inline-block px-2 py-1 text-xs rounded-full font-medium ${getStatusColor(metric.status)}`}>
                  {metric.status}
                </span>
              </div>
            </div>
          ))}
        </div>

        <div className="space-y-4">
          <div className="p-4 bg-green-50 border border-green-200 rounded-lg">
            <h4 className="font-medium text-green-900 mb-2">Data Protection Measures</h4>
            <ul className="text-sm text-green-800 space-y-1">
              <li>• AES-256 encryption for data at rest</li>
              <li>• TLS 1.3 for data in transit</li>
              <li>• Role-based access control (RBAC)</li>
              <li>• Data masking in non-production environments</li>
            </ul>
          </div>
          
          <div className="p-4 bg-blue-50 border border-blue-200 rounded-lg">
            <h4 className="font-medium text-blue-900 mb-2">Privacy Controls</h4>
            <ul className="text-sm text-blue-800 space-y-1">
              <li>• Customer consent management system</li>
              <li>• Right to be forgotten implementation</li>
              <li>• Data minimization principles enforced</li>
              <li>• Regular privacy impact assessments</li>
            </ul>
          </div>
        </div>
      </div>
    </div>
  );
}
